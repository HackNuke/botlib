# basic set

>>> from bot.obj import Object, set
>>> o = Object()
>>> set(o, "key", "value")
>>> o.key
'value'

# attribute set

>>> from bot.obj import Object
>>> o = Object()
>>> o.key = "value"
>>> o.key
'value'

# save/load

>>> from bot.obj import Object, load, save, setwd
>>> setwd(".test")
>>> o = Object()
>>> o["key"] = "value"
>>> p = save(o)
>>> p
'bot.obj.Object/...
>>> oo = Object()
>>> load(oo, p)
>>> oo.key
'value'
