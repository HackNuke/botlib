# basic set

>>> from bot import Object
>>> o = Object()
>>> o.set("key", "value")
>>> o.key
'value'

# attribute set

>>> from bot import Object
>>> o = Object()
>>> o.key = "value"
>>> o.key
'value'

# save/load

>>> import bot
>>> bot.wd = ".test"
>>> o = bot.Object()
>>> o["key"] = "value"
>>> p = o.save()
>>> p
'bot.Object/...
>>> oo = bot.Object()
>>> oo.load(p)
{"key": "value"}
>>> oo.key
'value'
