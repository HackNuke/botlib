# basic set

>>> from obj import Object, set
>>> o = Object()
>>> set(o, "key", "value")
>>> o.key
'value'

# attribute set

>>> from obj import Object
>>> o = Object()
>>> o.key = "value"
>>> o.key
'value'

# save/load

>>> import obj
>>> obj.wd = ".test"
>>> o = Object()
>>> o["key"] = "value"
>>> p = obj.save(o)
>>> p
'obj.Object/...
>>> oo = obj.Object()
>>> obj.load(oo, p)
>>> oo.key
'value'
